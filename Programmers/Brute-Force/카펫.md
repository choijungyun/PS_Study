/*
갈색격자의 수 : brown
노란색 격자의 수 : yellow
카펫의 가로, 세로 크기

중앙은 노란색, 테두리 1줄은 갈색
모양그려서 생각

*/

class Solution {
    public int[] solution(int brown, int yellow) {
        int[] answer = new int[2];
        
        //brown + yellow <= 2005000
        for(int i=1; i<=brown+yellow; i++){
            if((brown+yellow)%i==0){
                if((i-2)*(((brown+yellow)/i)-2)==yellow){
                    answer[0] = (brown+yellow)/i;
                    answer[1] = i;
                    break;
                }
            }
        }
        return answer;
    }
}
